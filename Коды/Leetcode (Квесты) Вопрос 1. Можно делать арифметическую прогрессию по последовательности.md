```` python
class Solution:
    def canMakeArithmeticProgression(self, arr: List[int]) -> bool:
        arr = sorted(arr) #сортировка исходного массива
        result = True #результат выполнения функци - по умолчанию
        arr_len = len(arr) #длина массива arr
        diff = arr[0] - arr[1] #разница между первым и вторым элементом отсортированного списка
        diff2 = 0

        if arr_len == 2:
            result = True
            return result
        
        
        
        for i in range(1, arr_len):
            
            if i + 1 != arr_len:
                diff2 = arr[i] - arr[i + 1]
            
            if diff != diff2:
                result = False
                return result

            
        result = True 
        return result


````

Описание:
	Сначала мы создаем класс, далее задаём функцию, в которой и будем работать. 
	1) Сортируем исходный массив
	2) Задаем результат выполнения функции по умолчанию
	3) Присваиваем в переменную длину массива arr
	4) Задаём в переменную diff - разницу между первым и вторым элементом сортированного списка
	5) Задаем переменную diff2, которая по умолчанию ровна 0
	6) Делаем первую проверку, если длина списка равна двум, то есть всего состоит из двух чисел, то выдаём True
	7) Далее перебираем для переменной i с длинной от первого символа до конца списка
	8) Делаем ещё одну проверку внутри цикла for, если следующий индекс не равен длине массива, то если условие выполняется, вычисляется разница, избегая выход за границы
	9) Следующая проверка, если разница1 не равна разнице2, то возвращаем False 
	10) Если цикл прошёл и не заметил отклонений, то под конец выполняется result = True и возвращается результат
