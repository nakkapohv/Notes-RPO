class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        for i in range(len(nums)):
            for j in range (i + 1, len(nums)):
                if nums[i] + nums[j] == target:
                    return[i, j]
        return
        

````

Создаём класс Soluction, далее создаем функцию(метод) twoSum внутри класса Soluction и передаём туда self (указатель ссылки), переменную nums предаём ей тип данных - список целых чисел и переменную target которой присваиваем тип данных integer (целые числа), далее используем -> (говорим функции что она должна возвращать список целых чисел). Цикл for перебирает все значения из последовательности целых чисел range для переменной i и в скобках range(len(nums)) возвращаем длину объекта (количество nums), далее задаём цикл for, который перебирает все значения из последовательности range для переменной j, начиная c i+1 и до конца списка (не включая последний индекс), далее строка if nums[i] + nums[j] == target: (проверяет равна ли сумма двух элементов списка заданному целому значению target, если это является True, то далее выполняется блок кода внутри if), где мы возвращаем значения i и j, а далее возвращаем и весь цикл for с самого начала, задача решена.

class Solution:
    def isPalindrome(self, x: int) -> bool:

        z = []
        for i in str(x):
            z.append(i)
        if z[0] == "-":
            return False
        if len(z) % 2 == 0:
            return False
        else:
            if z[0] == z[-1]:
                return True
            else: 
                return False
````
