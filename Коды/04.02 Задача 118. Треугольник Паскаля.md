class Solution:
    def generate(self, numRows: int) -> List[List[int]]:

        triangle = [[1]] #Создаём первую строку и она всегда будет 1

        for i in  range(1, numRows): #Цикл идёт с i=1 до i=numsRows-1, для всех последующих строк, начиная со второй

            row = [1] #каждая строка начинается с 1 

            for j in range(1, i): #Перебираем идексы j, кроме первых и последних, т.к они всегда равны
                num = triangle[i - 1][j - 1] + triangle[i - 1][j]
                #triangle[i - 1][j - 1] - это элемент слева сверху
                #triangle[i - 1][j] - это элемент справа сверху
                row.append(num) #После вычисления суммы, она добавляется в текущую строку

            row.append(1) #В конечно строки добволяем 1, так как каждя строка заканчвает на 1 
            triangle.append(row) #Добовляем строку row в список triangle
        return triangle #Возращаем полный треугольник
