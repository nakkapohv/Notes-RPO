import random, time
import matplotlib.pyplot as mplb

def merge_sort(A):
    A_copy = A.copy()
    
    if len(A_copy) <= 1:
        return A_copy
        
    mid = len(A_copy) // 2
    left = merge_sort(A_copy[:mid])
    right = merge_sort(A_copy[mid:])
    
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
            
    result.extend(left[i:])
    result.extend(right[j:])
    
    return result

sort_func = merge_sort


def rand_list(num):
    l = [random.randint(0, 100) for _ in range(num)]
    return l

arrays = []
nums = 10

for i in range(11):
    arrays.append(rand_list(int(nums)))
    if i == 0: nums *= 35
    if i < 3: nums *= 1.5
    elif i < 7: nums *= 1.3
    else: nums *= 1.2


print('Начало')
for _ in range(100):
    for arr in arrays[:3]:
        sort_func(arr)
print('Конец')

sizes = []
avg_times = []
iteration = 0

for arr in arrays:
    average_time = 0
    iteration += 1
    
    for _ in range(10):
        start_time = time.perf_counter()
        A_sorted = sort_func(arr)
        end_time = time.perf_counter()
        execution_time = end_time - start_time
        average_time += execution_time
        
    avg_time = average_time / 10
    sizes.append(len(arr))
    avg_times.append(avg_time)
    
    print('Закончен подсчёт сортировки последовательности длиной:', len(arr), f"\t{avg_time:.6f} сек.", f"\tИтерация: {iteration}")

mplb.figure(figsize=(10, 8))
mplb.plot(sizes, avg_times, marker='o')
mplb.title(f'Сортировка: {sort_func.__name__}')
mplb.xlabel('Длина')
mplb.ylabel('Среднее время, сек')
mplb.grid(True)
mplb.tight_layout()

mplb.savefig(f"{sort_func.__name__}_graph.png", dpi=300, bbox_inches='tight')
print(f"Результат сортировки сохранен в: {sort_func.__name__}_graph.png")
mplb.close()
