import random, time
def bubble_sort(A):
 A_copy = A.copy()
 n = len(A_copy)
 for i in range(n-1):
 swapped = False
 for j in range(n-1-i):
 if A_copy[j] > A_copy[j + 1]:
 A_copy[j], A_copy[j + 1] = A_copy[j + 1], A_copy[j]
 swapped = True
 if not swapped:
 break
 return A_copy
